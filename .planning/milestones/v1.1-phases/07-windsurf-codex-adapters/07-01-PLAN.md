pub trait ToolAdapter: Send + Sync {
    fn name(&self) -> ToolKind;
    fn display_name(&self) -> &str;
    fn native_instruction_path(&self) -> &str;
    fn conditional_tags(&self) -> &[&str] { &[] }
    fn gitignore_entries(&self) -> Vec<String> { vec![] }
    fn watch_paths(&self) -> Vec<&str> { vec![self.native_instruction_path()] }
    fn default_sync_strategy(&self) -> SyncStrategy { SyncStrategy::Symlink }
    fn detect(&self, project_root: &Path) -> Result<DetectionResult, AisyncError>;
    fn read_instructions(&self, project_root: &Path) -> Result<Option<String>, AisyncError>;
    fn plan_sync(&self, project_root: &Path, canonical_content: &str, strategy: SyncStrategy) -> Result<Vec<SyncAction>, AisyncError>;
    fn sync_status(&self, project_root: &Path, canonical_hash: &str, strategy: SyncStrategy) -> Result<ToolSyncStatus, AisyncError>;
    fn plan_memory_sync(&self, project_root: &Path, memory_files: &[PathBuf]) -> Result<Vec<SyncAction>, AisyncError>;
    fn translate_hooks(&self, hooks: &HooksConfig) -> Result<HookTranslation, AisyncError>;
}
