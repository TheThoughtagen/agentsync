    pub struct MemoryEngine;

    impl MemoryEngine {
        /// List all .md files in .ai/memory/, sorted alphabetically.
        pub fn list(project_root: &Path) -> Result<Vec<PathBuf>, AisyncError>;

        /// Create a new memory file .ai/memory/<topic>.md with "# <Topic>" header.
